快速幂的应用

2022/11/16 数论 逆元 矩阵运算

快速幂应用的推广

$a^b$

来简单复习一下快速幂
对于一个要求计算 的题目， ，若用朴素算法的复杂度是O(b)

    int ans = a;
    for(int i = 1; i <= b; ++i){
      ans = ans * a;
    }

但倘若我们能让b二进制分解的话，可以优化该复杂度
假设当b = 8时，b的二进制表示是 1000，即可以表示为

    int ans = 1;
    while(b){
      a *= a;
      b >>= 1;
    }
    ans = ans * a;

因此我们只需计算b二进制表达式下的每一个1到答案即可

int qm(int a, int b){
      int ans = 1;
      while(b){
        if(b & 1) ans *= a;
        a *= a;
        b >>= 1;
      }
      return ans;
}

在取模p的情况下

int qm(int a, int b, int p){
      int ans = 1;
      while(b){
        if(b & 1) ans = ans * a % p;
        a = a * a % p;
        b >>= 1;
      }
      return ans;
}

void mul(int a[], int b[][2]){
    int c[2] = {0};

    c[1] = a[0] * b[1][0] + a[1] * b[1][1];
    a[0] = c[0] % p;

void cal(int a[][2], int b[][2]){
    int c[2][2] = {0};
    rep(i, 0, 1){
        rep(j, 0, 1){
            rep(k, 0, 1){
                c[i][j] += a[i][k] * b[k][j] % p;
            }
        }
    }
    rep(i, 0, 1){
        rep(j, 0, 1){
            a[i][j] = c[i][j] % p;
        }
    }
}
